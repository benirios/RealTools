function makeSupabaseMockForUpsert(existingScoreVersion = null, savedRow = null) {
  let upsertCallCount = 0
  return {
    from(table) {
      if (table === 'opportunity_scores') {
        return {

                error: null,
              }),
              single: async () => ({ data: savedRow, error: null }),

                }
              },
            }
          },
        }
      }
      return {}
    },
  }
